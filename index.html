Code example

Here's some TypeScript code from Angular + Firebase project. Not so complex as it seems, but it's not codewars catas after all. 
                    
class Queries {
  data: string;
  datetime: string;
  id?: string;
}

@Component({
  selector: 'app-main',
  templateUrl: './main.component.html',
  styleUrls: ['./main.component.css']
})

export class MainComponent implements OnInit {
public queriesCollection: AngularFirestoreCollection;
public queries: Observable;
public searchRequest: FormControl;

constructor( public db: AngularFirestore ) { }

ngOnInit() {
this.searchRequest = new FormControl('', [Validators.required , Validators.maxLength(100)]);
this.queriesCollection = this.db.collection('Queries', ref => ref.orderBy('datetime', 'desc').limit(5));
this.queries = this.queriesCollection.snapshotChanges().pipe(map(items => items.map(a => {
      const data = a.payload.doc.data();
      data.id = a.payload.doc.id;
      return data;
      })));
}
